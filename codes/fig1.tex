\begin{tikzpicture}
% === PARÁMETROS ===
\definecolor{colorLinea}{HTML}{2365A6}
\definecolor{colorAnomalia}{HTML}{FFCCCC}
\definecolor{colorScore}{HTML}{808080}
\definecolor{colorUmbral}{HTML}{FF0000}

\pgfplotsset{
    estiloPlot/.style={width=10cm, height=2.5cm, axis lines*=left, grid=major, 
    grid style={dashed, gray!30}, xticklabels={,,}},
    estiloAnomalia/.style={area legend, fill=colorAnomalia, draw=none}
}

% === ESTRUCTURA PRINCIPAL ===
\begin{groupplot}[group style={group size=1 by 4, vertical sep=0.5cm}, 
                  xmin=0, xmax=100, xmajorgrids=true]

    % Panel 1: Temperature
    \nextgroupplot[estiloPlot, ylabel={Temp (C)}]
    \fill[colorAnomalia] (axis cs:20,0) rectangle (axis cs:40,10);
    \fill[colorAnomalia] (axis cs:70,0) rectangle (axis cs:90,10);
    \addplot[color=colorLinea, domain=0:100, samples=200] {5 + 0.5*sin(deg(x/5)) + (and(x>20,x<40)*5) - (and(x>70,x<90)*5)};

    % Panel 2: Voltage
    \nextgroupplot[estiloPlot, ylabel={Volt (V)}]
    \fill[colorAnomalia] (axis cs:20,0) rectangle (axis cs:40,10);
    \fill[colorAnomalia] (axis cs:70,0) rectangle (axis cs:90,10);
    \addplot[color=colorLinea, domain=0:100, samples=200] {5 + 0.5*cos(deg(x/5)) - (and(x>20,x<40)*5) + (and(x>70,x<90)*5)};

    % Panel 3: Attitude
    \nextgroupplot[estiloPlot, ylabel={Att (deg)}]
    \fill[colorAnomalia] (axis cs:20,-2) rectangle (axis cs:40,2);
    \fill[colorAnomalia] (axis cs:70,-2) rectangle (axis cs:90,2);
    \addplot[color=colorLinea, domain=0:100, samples=200] {0.5*sin(deg(x/10))};

    % Panel 4: Anomaly Score
    \nextgroupplot[estiloPlot, ylabel={Score}, xlabel={Time (s)}, xticklabels={0,20,40,60,80,100}]
    % ASUNCIÓN: Representación esquemática de picos de anomalía
    \addplot[color=colorScore, domain=0:100, samples=200] {0.5 + 4*exp(-(x-30)^2/5) + 4*exp(-(x-80)^2/5)};
    \draw[color=colorUmbral, dashed, thick] (axis cs:0,2) -- (axis cs:100,2);

\end{groupplot}

% === ANOTACIONES ===
\node[anchor=south, font=\bfseries] at ($(group c1r1.north)!0.5!(group c1r1.north) + (0,0.5)$) {Multivariate System Anomaly Detection};
\end{tikzpicture}
